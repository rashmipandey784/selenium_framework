pipeline {
    agent any

    stages {

        stage('Checkout') {
            steps {
                git 'https://github.com/rashmipandey784/selenium_framework.git'
            }
        }

        stage('Verify Python') {
            steps {
                bat '"C:\\Program Files\\python.exe" --version'
            }
        }

        stage('Install Dependencies') {
            steps {
                bat '"C:\\Program Files\\python.exe" -m pip install -r requirement.txt'
            }
        }

        stage('Run Tests') {
            steps {
                bat '"C:\\Program Files\\python.exe" -m pytest -v -s'
            }
        }
    }

    post {
        always {
            echo 'Pipeline finished'
        }
    }
}
